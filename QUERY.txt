CREATE TABLE ELEGIVEIS AS SELECT 
BASE_DISCIPLINAS.CODIGO, BASE_DISCIPLINAS.DISCIPLINA, BASE_DISCIPLINAS.CREDITOS, 
BASE_DISCIPLINAS.RECOMENDACAO1, BASE_DISCIPLINAS.RECOMENDACAO2, 
BASE_DISCIPLINAS.RECOMENDACAO3, BASE_DISCIPLINAS.RECOMENDACAO4, BASE_DISCIPLINAS.TIPO
FROM BASE_DISCIPLINAS INNER JOIN OFERTA_DISCIPLINAS ON
'%' + BASE_DISCIPLINAS.CODIGO + '%' LIKE OFERTA_DISCIPLINAS.TEXTO;
